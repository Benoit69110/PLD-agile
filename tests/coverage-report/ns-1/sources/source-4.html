529      * Find all shortest path from a starting point
530      * @param startIntersection
531      * @return pi the precedence table, for each intersection we have the shortest segment to go to another intersection
532      */
533     public HashMap<Intersection,Segment> dijkstra(Intersection startIntersection){
534         HashMap<Intersection,Double> d = new HashMap<>();
535         HashMap<Intersection,Segment> pi = new HashMap<>();
536         ArrayList<Intersection> blanc= new ArrayList<>(),noir = new ArrayList<>();
537         PriorityQueue<Intersection> gris = new PriorityQueue<>(Comparator.comparingDouble(d::get));
538         graphe.forEach((i, dest) -> {
539             if(i.equals(startIntersection)){
540                 d.put(startIntersection,0.0);
541             }else {
542                 d.put(i, Double.MAX_VALUE);
543             }
544             pi.put(i,null);
545             blanc.add(i);
546         });
547         gris.add(startIntersection);
548         while (!gris.isEmpty()){
549             Intersection minimalSuccessor = gris.peek();
550             LinkedList<Segment> destinations = graphe.get(minimalSuccessor);
551             destinations.forEach((segment)->{
552                 Intersection successor = segment.getDestination();
553                 if((blanc.contains(successor)) || (gris.contains(successor))){
554                     if(d.get(successor) > d.get(minimalSuccessor) + segment.getLength()){
555                         d.put(successor, d.get(minimalSuccessor) + segment.getLength());
556                         pi.replace(successor, segment);
557                     }
558                 }
559                 if(blanc.contains(successor)){
560                     gris.add(successor);
561                     blanc.remove(successor);
562                 }
563             });
564             noir.add(minimalSuccessor);
565             gris.remove(minimalSuccessor);
566         }
567         return pi;
568     }
